import engine.*; import engine.collision.*; import java.util.List; public class MyGame { public static void main(String[] args) { // Create game and scene Game game = new Game("My Awesome Game"); Scene scene = new Scene("Main Scene"); game.addScene(scene); game.setCurrentScene(scene); // Create player with circle collider Circle playerCollider = new Circle(new Vector2D(400, 300), 25); GameObject player = new GameObject("player", playerCollider); // Add physics component PhysicsComponent physics = new PhysicsComponent(); physics.setMass(1.0); physics.setUseGravity(true); player.addComponent(physics); // Add to scene scene.addGameObject(player); // Start game loop game.start(); game.gameLoop(); } }

üìñ Step-by-Step Tutorial

Learn how to build a complete 2D game from scratch using this engine. Each lesson builds upon the previous one.

Lesson 1: Your First Game Object

Every game starts with a GameObject. Think of GameObjects as the "actors" in your game world. They can be players, enemies, platforms, or anything else that exists in your game.

Java - Lesson 1
// First, import the engine packages import engine.*; import engine.collision.*; public class Lesson1_FirstObject { public static void main(String[] args) { // Step 1: Create a Circle collider // The collider defines the object's shape for collision detection Vector2D position = new Vector2D(400, 300); double radius = 25; Circle collider = new Circle(position, radius); // Step 2: Create the GameObject with the collider GameObject player = new GameObject("player", collider); // Step 3: Set initial properties player.setPosition(new Vector2D(400, 300)); player.setVelocity(new Vector2D(50, 0)); // Moving right! System.out.println("Player created at: " + player.getPosition()); System.out.println("Player velocity: " + player.getVelocity()); } } // üìö What you learned: // ‚Ä¢ Vector2D - Represents 2D positions and velocities // ‚Ä¢ Circle - A circular collision shape // ‚Ä¢ GameObject - The base entity in your game

Lesson 2: Adding Physics

Physics makes your game feel real! The PhysicsComponent adds gravity, velocity, and collision response to your objects.

Java - Lesson 2
// Continuing from Lesson 1... // Add physics to make the object fall PhysicsComponent physics = new PhysicsComponent(); // Mass affects how objects respond to forces physics.setMass(1.0); // Enable gravity - objects will fall down physics.setUseGravity(true); // Restitution = bounciness (0 = no bounce, 1 = full bounce) physics.setRestitution(0.7); // Friction slows down movement on surfaces physics.setFriction(0.1); // Add the physics component to our player player.addComponent(physics); // Now let's update the player position over time for (int frame = 0; frame < 5; frame++) { double deltaTime = 0.016; // 60 FPS = 16ms per frame // This updates velocity (applies gravity) and position physics.update(deltaTime); System.out.printf("Frame %d: pos=%s, vel=%s%n", frame, player.getPosition(), player.getVelocity()); } // üìö What you learned: // ‚Ä¢ PhysicsComponent adds realistic movement // ‚Ä¢ Gravity pulls objects downward // ‚Ä¢ Restitution controls bounciness // ‚Ä¢ Friction slows sliding objects

Lesson 3: Collision Detection

Collision detection is how your game knows when two objects touch. The engine handles this automatically!

Java - Lesson 3
// Create two objects that will collide // Player - a moving circle Circle playerShape = new Circle(new Vector2D(100, 100), 30); GameObject player = new GameObject("player", playerShape); // Enemy - a stationary rectangle Rectangle enemyShape = new Rectangle(new Vector2D(180, 100), 60, 40); GameObject enemy = new GameObject("enemy", enemyShape); // Simple collision check - are they touching? boolean isColliding = player.collidesWith(enemy); System.out.println("Initial collision: " + isColliding); // Move player towards enemy player.setPosition(new Vector2D(140, 100)); // Check again isColliding = player.collidesWith(enemy); System.out.println("After moving: " + isColliding); // Get detailed collision information CollisionResult result = player.getCollisionWith(enemy); if (result.isColliding()) { System.out.println("üí• COLLISION DETECTED!"); System.out.println("Penetration depth: " + result.getPenetrationDepth()); System.out.println("Contact point: " + result.getContactPoint()); } // üìö What you learned: // ‚Ä¢ collidesWith() - Simple true/false check // ‚Ä¢ getCollisionWith() - Detailed collision data // ‚Ä¢ CollisionResult - Contains penetration depth and contact point

Lesson 4: Building a Complete Scene

A Scene is like a game level or screen. It holds all your game objects and manages updates.

Java - Lesson 4
// Create a complete game scene // Step 1: Create the Game Game game = new Game("My Platformer"); // Step 2: Create a Scene (like a level) Scene level1 = new Scene("Level 1"); // Step 3: Add the scene to the game game.addScene(level1); game.setCurrentScene(level1); // Step 4: Create and add game objects // Create player GameObject player = new GameObject("player", new Circle(new Vector2D(100, 300), 20)); player.addComponent(new PhysicsComponent()); level1.addGameObject(player); // Create ground platform GameObject ground = new GameObject("ground", new Rectangle(new Vector2D(400, 550), 800, 50)); PhysicsComponent groundPhysics = new PhysicsComponent(); groundPhysics.setStatic(true); // Static objects don't move ground.addComponent(groundPhysics); level1.addGameObject(ground); // Create some platforms String[] platformPositions = {"200,400", "400,300", "600,200"}; for (int i = 0; i < platformPositions.length; i++) { String[] pos = platformPositions[i].split(","); double x = Double.parseDouble(pos[0]); double y = Double.parseDouble(pos[1]); GameObject platform = new GameObject("platform_" + i, new Rectangle(new Vector2D(x, y), 150, 20)); PhysicsComponent p = new PhysicsComponent(); p.setStatic(true); platform.addComponent(p); level1.addGameObject(platform); } System.out.println("Scene created with " + level1.getGameObjectCount() + " objects"); // üìö What you learned: // ‚Ä¢ Game holds multiple Scenes // ‚Ä¢ Scenes contain GameObjects // ‚Ä¢ Static objects don't move but can be collided with // ‚Ä¢ You can add multiple objects to a scene

Lesson 5: Input and Player Control

Now let's make our player controllable! We'll use the InputComponent and PlayerController.

Java - Lesson 5
// Create a controllable player // Method 1: Using PlayerController (recommended for platformers) GameObject player = new GameObject("player", new Circle(new Vector2D(100, 300), 20)); // Add physics PhysicsComponent physics = new PhysicsComponent(); physics.setMass(1.0); physics.setUseGravity(true); player.addComponent(physics); // Add player controller with movement speed and jump force PlayerController controller = new PlayerController(200.0, 400.0); player.addComponent(controller); // Method 2: Manual keyboard control InputComponent customInput = new InputComponent() { private double speed = 150; @Override protected void onSpecificInputEvent(InputEvent event) { if (event instanceof KeyEvent) { KeyEvent key = (KeyEvent) event; if (key.getType() == InputEvent.Type.KEY_PRESSED) { // WASD or Arrow keys for movement double vx = 0, vy = 0; if (key.is(KeyEvent.KeyCode.W) || key.is(KeyEvent.KeyCode.UP)) { vy = -speed; } else if (key.is(KeyEvent.KeyCode.S) || key.is(KeyEvent.KeyCode.DOWN)) { vy = speed; } else if (key.is(KeyEvent.KeyCode.A) || key.is(KeyEvent.KeyCode.LEFT)) { vx = -speed; } else if (key.is(KeyEvent.KeyCode.D) || key.is(KeyEvent.KeyCode.RIGHT)) { vx = speed; } gameObject.setVelocity(new Vector2D(vx, vy)); } } } }; // Add custom input to another object GameObject npc = new GameObject("npc", new Circle(new Vector2D(300, 200), 15)); npc.addComponent(customInput); // üìö What you learned: // ‚Ä¢ PlayerController handles WASD + Arrow keys + Space for jump // ‚Ä¢ InputComponent lets you create custom input behavior // ‚Ä¢ You can override onSpecificInputEvent for custom controls

Lesson 7: The Game Loop

The game loop is the heartbeat of your game. It repeatedly updates the game state and renders frames.

Java - Lesson 7
// The complete game setup with game loop public class CompleteGame { public static void main(String[] args) { // 1. Create the game Game game = new Game("My Platformer"); // 2. Set target FPS (60 is standard) game.setTargetFPS(60); // 3. Create and setup scene Scene scene = new Scene("Level 1"); game.addScene(scene); game.setCurrentScene(scene); // 4. Add game objects setupLevel(scene); // 5. Start the game (initializes input, etc.) game.start(); // 6. Run the game loop System.out.println("üéÆ Starting game loop..."); game.gameLoop(); } private static void setupLevel(Scene scene) { // Create player GameObject player = new GameObject("player", new Circle(new Vector2D(100, 300), 20)); PhysicsComponent physics = new PhysicsComponent(); physics.setMass(1.0); physics.setUseGravity(true); player.addComponent(physics); PlayerController controller = new PlayerController(200, 400); player.addComponent(controller); scene.addGameObject(player); // Create ground GameObject ground = new GameObject("ground", new Rectangle(new Vector2D(400, 550), 800, 50)); PhysicsComponent groundPhysics = new PhysicsComponent(); groundPhysics.setStatic(true); ground.addComponent(groundPhysics); scene.addGameObject(ground); } } // What happens in gameLoop(): // 1. Calculate deltaTime (time since last frame) // 2. Update input system (keyboard, mouse) // 3. Update all game objects (physics, movement) // 4. Check all collisions // 5. Wait to maintain target FPS // 6. Repeat! // üìö What you learned: // ‚Ä¢ Game.gameLoop() handles everything automatically // ‚Ä¢ targetFPS controls how fast the game runs // ‚Ä¢ Everything updates in sequence each frame

Lesson 8: Finding Objects and Collision Events

Learn how to find specific objects and respond to collisions for game logic.

Java - Lesson 8
// Finding objects and handling collisions // Method 1: Find by ID GameObject player = scene.findGameObjectById("player_1"); if (player != null) { System.out.println("Found player!"); } // Method 2: Find by tag // First, add tags to objects GameObject coin = new GameObject("coin_1", new Circle(new Vector2D(200, 100), 10)); coin.addComponent(new TagComponent("coin", "collectible", "valuable")); scene.addGameObject(coin); // Find all coins in the scene List<GameObject> coins = scene.findGameObjectsByTag("coin"); System.out.println("Found " + coins.size() + " coins"); // Method 3: Check collisions each frame public void checkCollisions(Scene scene) { // Get all active collisions List<CollisionResult> collisions = scene.getAllCollisions(); for (CollisionResult collision : collisions) { System.out.println("üí• Collision detected!"); System.out.println("Depth: " + collision.getPenetrationDepth()); } } // Method 4: Get collisions for specific object public void checkPlayerCollisions(GameObject player, Scene scene) { List<GameObject> colliding = scene.getCollisionsWith(player); for (GameObject other : colliding) { // Check what we hit TagComponent tags = other.getComponent(TagComponent.class); if (tags != null && tags.hasTag("coin")) { System.out.println("ü™ô Collected a coin!"); scene.removeGameObject(other); // Remove the coin } else if (tags != null && tags.hasTag("enemy")) { System.out.println("üëπ Hit an enemy! Take damage."); } } } // üìö What you learned: // ‚Ä¢ findGameObjectById() - Find by unique ID // ‚Ä¢ findGameObjectsByTag() - Find all objects with a tag // ‚Ä¢ getAllCollisions() - Get all collisions in scene // ‚Ä¢ getCollisionsWith() - Get collisions for one object // ‚Ä¢ Tags help categorize and find objects easily
